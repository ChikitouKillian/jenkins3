pipeline {
    agent any

    environment {
        SONARQUBE_SERVER = 'SonarQube'
        SONAR_TOKEN = credentials('sonar-token-jenkin')
    }

    stages {
        stage('Analyse SonarQube') {
            steps {
                checkout scm
                withSonarQubeEnv(SONARQUBE_SERVER) {
                    sh 'mvn clean sonar:sonar -Dsonar.login=$SONAR_TOKEN'
                }
            }
        }
    }
}
